SELECT
	MODVehType as VehType,
	MODNatCode as TYPModCd,
	MODMakCD as schwackeCode,
	MODName as name,
	MODName2 as name2,
	MODModelSerCode as serialCode,
	CAST(MODBegin as INT) as yearBegin,
	CAST(MODEnd as INT) as yearEnd,
	TO_DATE(MODImpBegin, 'yyyyMM') as productionBegin,
	TO_DATE(MODImpEnd, 'yyyyMM') as productionEnd
FROM {DB_NAME}.model
WHERE data_date_part='{CURRENT_DATE_%Y-%m-%d}' AND
	data_timestamp_part='{CURRENT_TIMESTAMP}';
